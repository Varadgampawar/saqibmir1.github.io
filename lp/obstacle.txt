from time import sleep

while(True):
   
    i=GPIO.input(33)
    sleep(1)
    if(i==0):
        GPIO.output(18,GPIO.LOW)
        print("Not detected")
       
       
    else:
        GPIO.output(18,GPIO.HIGH)
        print("Detected")
